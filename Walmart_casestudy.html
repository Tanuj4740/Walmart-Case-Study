import numpy as np

# Function to calculate confidence interval
def calculate_ci(data, confidence_level=0.95):
    # Calculate sample mean and standard error
    sample_mean = np.mean(data)
    std_error = np.std(data, ddof=1) / np.sqrt(len(data))

    # Calculate the critical value based on the confidence level
    if confidence_level == 0.90:
        z_score = 1.645
    elif confidence_level == 0.95:
        z_score = 1.96
    elif confidence_level == 0.99:
        z_score = 2.576
    else:
        raise ValueError("Unsupported confidence level. Please choose from 0.90, 0.95, or 0.99.")

    # Calculate margin of error
    margin_of_error = z_score * std_error

    # Calculate confidence interval
    ci_lower = sample_mean - margin_of_error
    ci_upper = sample_mean + margin_of_error

    return ci_lower, ci_upper

# Set confidence levels
confidence_levels = [0.90, 0.95, 0.99]

# Set sample sizes
sample_sizes = [50, 100, 200, 500, 1000]

# Initialize dictionaries to store results
ci_results_female = {}
ci_results_male = {}

# Loop through sample sizes and confidence levels
for sample_size in sample_sizes:
    for confidence_level in confidence_levels:
        # Sample from the original data for female and male customers
        sample_female = np.random.choice(female_customers['Purchase'], size=sample_size, replace=True)
        sample_male = np.random.choice(male_customers['Purchase'], size=sample_size, replace=True)

        # Calculate confidence intervals using CLT
        ci_lower_female, ci_upper_female = calculate_ci(sample_female, confidence_level)
        ci_lower_male, ci_upper_male = calculate_ci(sample_male, confidence_level)

        # Store results
        ci_results_female[(sample_size, confidence_level)] = (ci_lower_female, ci_upper_female)
        ci_results_male[(sample_size, confidence_level)] = (ci_lower_male, ci_upper_male)

# Print results
print("Confidence Intervals for Average Spending of Female Customers:")
for key, value in ci_results_female.items():
    print("Sample Size:", key[0], "| Confidence Level:", key[1], "| Confidence Interval:", value)

print("\nConfidence Intervals for Average Spending of Male Customers:")
for key, value in ci_results_male.items():
    print("Sample Size:", key[0], "| Confidence Level:", key[1], "| Confidence Interval:", value)

# Function to calculate confidence interval using bootstrapping
def calculate_ci_bootstrap(data, n_bootstrap, confidence_level=0.95):
    # Initialize array to store sample means
    sample_means = np.zeros(n_bootstrap)

    # Generate bootstrap samples and calculate sample means
    for i in range(n_bootstrap):
        bootstrap_sample = np.random.choice(data, size=len(data), replace=True)
        sample_means[i] = np.mean(bootstrap_sample)

    # Calculate confidence interval
    alpha = 1 - confidence_level
    lower_percentile = alpha / 2 * 100
    upper_percentile = (1 - alpha / 2) * 100
    ci_lower = np.percentile(sample_means, lower_percentile)
    ci_upper = np.percentile(sample_means, upper_percentile)

    return ci_lower, ci_upper, sample_means

import numpy as np
import matplotlib.pyplot as plt

# Function to calculate confidence interval using bootstrapping
def calculate_ci_bootstrap(data, n_bootstrap, confidence_level=0.95):
    # Initialize array to store sample means
    sample_means = np.zeros(n_bootstrap)

    # Generate bootstrap samples and calculate sample means
    for i in range(n_bootstrap):
        bootstrap_sample = np.random.choice(data, size=len(data), replace=True)
        sample_means[i] = np.mean(bootstrap_sample)

    # Calculate confidence interval
    alpha = 1 - confidence_level
    lower_percentile = alpha / 2 * 100
    upper_percentile = (1 - alpha / 2) * 100
    ci_lower = np.percentile(sample_means, lower_percentile)
    ci_upper = np.percentile(sample_means, upper_percentile)

    return ci_lower, ci_upper, sample_means
